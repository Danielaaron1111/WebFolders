function toggleWishlist(productId) {
    // Use .includes() to check if product is in wishlist
    if (wishlist.includes(productId)) {
        // Use .findIndex() to locate the product
        const index = wishlist.findIndex(id => id === productId);
        // Use .splice() to remove it

function showActiveDeals() {
    // Step 1: Filter products with discount > 20%
    const activeDeals = productsWithDiscounts.filter(product => 
        product.discount > 20
    );
    
    // Step 2: Transform each deal with calculations
    const enrichedDeals = activeDeals.map(product => {
        const originalPrice = product.price;
        const discountAmount = originalPrice * (product.discount / 100);
        const discountedPrice = originalPrice - discountAmount;
        
        return {
            ...product,
            originalPrice,
            discountedPrice,
            discountAmount,
            savingsPercent: product.discount
        };

üìö Key Concepts Review

Array Methods

  • .map() - Transform elements ‚Üí new array
  • .filter() - Keep elements that pass test
  • .reduce() - Accumulate into single value
  • .forEach() - Execute function (no return)
  • .find() - Return first match
  • .findIndex() - Return index of match
  • .some() - True if ANY passes
  • .every() - True if ALL pass
  • .sort() - Sort in place (mutates)
  • .slice() - Extract section (no mutation)
  • .splice() - Add/remove (mutates)
  • .includes() - Check if contains value

DOM Techniques

  • createElement() - Create new elements
  • appendChild() - Add child to parent
  • removeChild() - Remove element
  • replaceChild() - Replace element
  • textContent - Set text (safe)
  • innerHTML - Set HTML (caution)
  • classList - Manage CSS classes
  • setAttribute() - Set attributes
  • addEventListener() - Event handlers

‚úÖ Best Practices

  • ‚úÖ Use textContent for user input (security)
  • ‚úÖ Use const for values that don't change
  • ‚úÖ Use arrow functions for callbacks
  • ‚úÖ Use template literals for strings
  • ‚úÖ Use destructuring to extract object properties
  • ‚úÖ Chain array methods for data pipelines
  • ‚úÖ Prefer createElement() over innerHTML
